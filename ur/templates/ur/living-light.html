آپ کی روح کی مافوق الفطرت زندہ روشنی خدا کی زندہ روشنی ہے۔


میں کیسے جان سکتا ھوں؟ مجھے اپنی روح کے اندر جھانکنے کی اجازت دی گئی ہے، اس کی توانائیاں اور قوتیں دکھائی جائیں گی، اور یہ سیکھنے کی اجازت دی گئی ہے کہ روح کا آسمان اور خدا سے براہ راست تعلق کیسے ہے۔ خدا زندہ روشنی ہے کیونکہ خدا موجود تمام مافوق الفطرت اور قدرتی روشنی کی حتمی تعریف ہے۔ تمام روشنیوں میں سے یہ سب سے بڑی روشنی آسانی سے دریافت کی جا سکتی ہے، محسوس کی جا سکتی ہے، اس تک رسائی حاصل کی جا سکتی ہے اور ایمان کے ذریعے دیکھی جا سکتی ہے کیونکہ یہ زندہ روشنی کسی کی زندہ روح کا جوہر ہے۔ ہماری روح خُدا کی اپنی روشنی اور سانس سے زندہ رہتی ہے۔ اس کے بغیر، کچھ بھی نہیں ہوسکتا ہے اور نہ ہی موجود ہوگا۔ خدا نور کی سب سے بڑی طاقت ہے، اور وہ اپنی روشنی ان تمام لوگوں کو دیتا ہے جو اس کی محبت کو اپنی روحوں میں داخل کرنے کے لیے تڑپتے ہیں جب معجزات اور انکشافات ہونے لگتے ہیں تاکہ یہ یقین دہانی کرائی جا سکے کہ ربط مضبوط ہو رہا ہے، اور اعلیٰ جہتوں تک رسائی کی اجازت دیتا ہے۔ جب خدا اور مسیح کو اپنے دلوں، روحوں اور روحوں میں مدعو کیا جاتا ہے، تو روحانی تبدیلیاں رونما ہونے لگتی ہیں کیونکہ جو کچھ آسمانی روشنی کا ہے وہ آپ کی اپنی روح کو برکت دیتا ہے۔ یہ تب ہوتا ہے جب آپ واقعی جانتے ہیں کہ آپ ان سب کے ساتھ اندر اور اوپر سے شامل ہیں۔ اور ایک بار شامل ہونے کے بعد، آپ کی زندہ روح سب کے سب سے بڑے، نہ ختم ہونے والے روحانی خزانوں کو حاصل کرنا شروع کر دیتی ہے۔ سچائی، روشنی، ایمان، محبت، روح اور روح کے ابدی اور لازوال رازوں کو سیکھنے کے لیے رابطے جو ہمیشہ آپ کے اپنے ہوں گے اور مسیح کے ذریعے ابدی ہونے کی ضمانت دی گئی ہے۔

آپ کی زندہ روح تمام روحانی اور مافوق الفطرت خزانوں میں سب سے امیر ہے، صبر کے ساتھ انتظار کر رہی ہے کہ آپ اس کے رازوں، اس کی طاقتوں، اس کی سچائیوں اور اس کے انعامات کو دریافت کریں۔ آپ کی روح آپ کے باطنی وجود کا بہت بنیادی، جوہر ہے، ہمیشہ کی زندگی کے مقصد کے لیے دی گئی ہے۔ آپ اپنی روشنی ہیں؛ آپ یہاں ہیں کیونکہ آپ کو آپ کی روح کا نور دیا گیا تھا تاکہ آپ دنیا کے ساتھ خدا کی روح القدس کو بانٹ سکیں۔ آپ چمکدار ہیں، آپ خوبصورت ہیں، آپ شاندار ہیں، اور آپ سے محبت کی جاتی ہے۔ تمام آسمان صبر سے آپ کا انتظار کر رہے ہیں کہ یہ سب معلوم ہو جائے۔ آپ کو اس دنیا میں خدا کے علاوہ کسی اور چیز کی ضرورت نہیں ہے تاکہ یہ سیکھ سکیں کہ فرشتے کہاں رہتے ہیں۔ آپ کی روح کے پاس آسمانی چیزوں سے متعلق تمام جوابات ہیں جن کے بارے میں آپ کو کبھی بھی سیکھنے کی ضرورت ہوگی۔ وہ آپ کو اپنے تمام اسرار سکھائیں گے اور آپ کو تمام روشنیوں میں سے روشن ترین روشنی کی طرف لے جائیں گے، جہاں جنت رہتی ہے۔ اپنی روح اور روح کو غور سے سننا سیکھیں تاکہ ان کی تمام خواہشات اور ضروریات پوری ہو سکیں۔ اپنی روح کو وہ سب کچھ دے دو جو وہ چاہتا ہے، اور یہ تمہیں بہت زیادہ اجر دے گا۔ اپنی روح کو پاکیزگی کی نعمتوں سے اچھی طرح پرورش کریں تاکہ اس کے راز افشا ہوں۔ روح کی روشنی ان تمام چیزوں سے ایندھن بننے کی بھوکی ہے جو مقدس اور سچی ہے۔ آپ جو ہیں اس پر بھروسہ کریں۔ کسی بھی چیز سے پہلے آپ اپنی زندہ روح ہیں۔ آپ کی روح ایک معجزہ ہے - صرف خدا کا کام۔ اندر جھانکیں، اور آپ کو وہ خزانہ نظر آئے گا جو آپ واقعی ہیں۔ آپ اکیلے تمام کائنات میں سب سے قیمتی خزانہ ہیں. اور روح کے پہلے سنہری اصول کو ہمیشہ یاد رکھیں: اس پر کوئی سوال یا شک نہیں کیا جا سکتا۔ یہ صرف بلا شبہ ایمان کے ذریعے ہی قبول کیا جا سکتا ہے کہ آپ کی اپنی تمام چیزوں کو برکت دی جائے جو اوپر آسمانوں سے برکت اور برکت حاصل کرتی ہے۔

میں نے روح کی روشنی کے بارے میں کیا سیکھا۔

ان لوگوں کے لیے جو روح میں رہتے ہیں، زندگی طاقت حاصل کرنے کے لیے روح کے حالات کو سیکھنے کے بارے میں ہے۔ جیسے جیسے ایمان بڑھتا ہے، اسی طرح زندہ روشنی کے بارے میں آگاہی بھی آتی ہے، حالانکہ اگر ایمان کی بنیاد راستبازی پر نہیں ہے، تو کسی کے ادراک کو آسانی سے کنٹرول کیا جا سکتا ہے جو کہ نہیں ہے۔ جو لوگ ایمان کے بارے میں سیکھتے ہیں وہ روشنی اور اندھیرے کے بارے میں سیکھتے ہیں، اور جو نہیں جانتے وہ دونوں سے اندھے ہو جاتے ہیں۔ میں خود، میں روح القدس کی روشنی سے سب سے زیادہ واقف ہو گیا ہوں، جو ان تمام لوگوں کو غیر مشروط طور پر فراہم کرتا ہے جو اپنے آپ سے جڑنا چاہتے ہیں۔ میں ہر سانس کے ساتھ خدا کی زندہ روشنی کی تعریف کرتا ہوں، کیونکہ جو کچھ مجھ میں ہے وہ اس کا اپنا ہے۔ اور اس وجہ سے کہ میرا ایمان مجھے کہاں لے گیا ہے، میں سب سے بڑھ کر اس کی زندہ روشنی پر یقین رکھتا ہوں۔ روح القدس کی زندہ روشنی سب سے زیادہ طاقتور قدرتی اور مافوق الفطرت روشنی ہے، اور یہ ان تمام لوگوں کے لیے اپنی محبت کو بانٹتی اور چمکاتی ہے جو اس کے پیاروں کی خواہش اور خواہش رکھتے ہیں۔ ہمارا خدا، ہمارا نجات دہندہ مسیح یسوع، اور روح القدس وہ ذرائع ہیں جہاں سے تمام غیر مشروط محبت، تمام ابدی روشنی، اور تمام اٹل ایمان پیدا ہوتے ہیں۔ خدا کی زندہ روشنی کے بارے میں جو کوئی سچ مانتا ہے وہ اپنے لئے سچ ہے جب امید شک نہیں کرتی ہے۔ اس کی روشنی ایک خوفناک روشنی ہے جو اس کی اپنی ذات میں ایمان کے ذریعے دیکھی اور محسوس کی جا سکتی ہے۔ اس کی زندہ روشنی زندہ روح کا نچوڑ ہے، جو سب کو یہ سیکھنے کے لیے دیا گیا ہے کہ کیسے پہچانا جائے اور اس سے سیکھا جائے۔

اپنے آپ کو پہچاننے اور سمجھنے کی کوشش کرنے سے پہلے خدا کے نور کو پہچاننا چاہیے۔ ہاں، ہر ایک کے پاس دریافت کرنے کے لیے اپنی روح کی روشنی ہوتی ہے۔ صرف ایمان کے ذریعے ہی وہ سب کچھ سکھایا جا سکتا ہے جو ان کی روشنی پر مشتمل ہے۔ ایک بار جب کوئی خدا کی مقدس ترین روشنیوں کے لیے جینا شروع کر دیتا ہے، تمام روحانی سچائیوں کے لیے بیداری شروع ہو جاتی ہے اور بڑھتی ہے۔ یہ سب کچھ ایمان کی طاقت کو سیکھنے کے بارے میں ہے، اور ایک بار بیداری بڑھنے کے بعد، غیر مرئی زندہ روشنی نظر آنا شروع ہو جاتی ہے۔ یہ وہی ہے جو روح القدس میں رہنا ظاہر کرتا ہے۔ وہ تمام لوگ جنہوں نے اپنے دلوں اور روحوں میں اس کے مقدس نور کا خیرمقدم کیا ہے وہ روشن ہو گئے ہیں، کیونکہ وہ خود زندگی کے جوہر کی سب سے بڑی حقیقت ہے اور ایک مومن کی روح کے ساتھ سب سے زیادہ مضبوط رشتہ ہے۔

اپنی روح کو سننے کے لیے اپنی روح کو سننا سیکھیں۔ اپنی روح کو سننے کے لیے اپنی روح کو سننا سیکھیں۔ اس کے بعد ہی کوئی یہ سیکھتا ہے کہ کس طرح باشعور تینوں کا ایک حصہ ہے جو کسی کی زندگی کی طاقت کو سمجھنے کا آغاز کرتا ہے۔ اپنے ایمان کو سننا سیکھنے کے لیے اپنی تینوں کو سننا سیکھیں۔ اور ایک بار جب ہر ایک کو پوری طرح سے سمجھ لیا جاتا ہے، یہ وہ وقت ہے جب اور کہاں چار روحانی قوتیں جو آپ کی زندگی کا تحفہ ہیں، سچی اور مکمل وجوہات کی بنا پر سیکھنا شروع کر دیتی ہیں کہ آپ کے زندگی کے مقاصد کیا ہیں۔ یہ وہ جگہ ہے جہاں آپ اپنی قسمت میں موجود تمام چیزوں کے ساتھ رابطہ اور رابطہ قائم کرنا شروع کرتے ہیں اور آپ کا ایمان اوپر کے آسمانوں سے کیسے اور کیوں جڑتا ہے۔ آپ کی زندہ روشنی زندہ ہے۔ آپ کا ایک روح کا تحفہ ابدی ہے، اور یہ آپ کے لیے اس کی سچائیوں کو جاننے کے لیے بہت بے چین ہے۔ جتنا آپ کر سکتے ہیں سیکھنے کے لیے وقت کی اہمیت ہے۔ اگر کوئی دیر سے سیکھے تو اس کی کتنی برکتیں دی جائیں اور بانٹیں؟ یا اگر کوئی زندہ روح کی سچائیوں کو نہیں جانتا تو اس کا کیا مطلب ہے کہ اس دنیا کو چھوڑنا ہے؟ براہ کرم اپنے روحانی مستقبل کو خطرے میں نہ ڈالیں۔ ہر شخص کے لیے سیکھنے کے لیے بہت سے اسباق ہیں جب وہ اس دنیا میں اپنا مقررہ وقت گزار رہا ہے۔ وقت کو گزرنے نہ دیں اب اپنی روحانی تقدیر کا حصہ بنیں، اور آپ کو برکت ملے گی۔

روح کی روشنی کی آگہی نے میری پوری زندگی بدل دی ہے۔ اس نے مجھے تمام روحانی سچائیوں کو تلاش کرنے کی ناقابل تردید تڑپ دی ہے جو میں کر سکتا ہوں۔ روحانی طور پر بڑھنے کے قابل ہونے کے لیے میں ابھی بہت کچھ سیکھنا چاہتا ہوں جس کی مجھے ایک جھلک مل گئی ہے۔ میں چاہتا ہوں کہ میرا مقدس رب مجھ سے راضی ہو جائے جو میری خواہش ہے کہ میں اپنی پوری طاقت سے اس کی خدمت کروں۔ روح کی بیداری نے مجھے ایک عقیدت مند روحانی آدمی میں بدل دیا ہے۔ میں سب کے اندر سچ کی تلاش کرتا ہوں۔ اور میری بیداری کی وجہ سے، اس نے مجھے بہت کچھ عطا کیا ہے جسے میں ابھی سمجھنا شروع کر رہا ہوں۔ اس سے کوئی فرق نہیں پڑتا کہ میں یہاں سے کہاں جاتا ہوں۔ کوئی چیز میری عقیدت کو نہیں روکے گی۔ میں نے ہمیشہ مسیح میں ایک رہنے کے لیے اپنی جان اور جان سپرد کر دی ہے۔ اس کی زندہ روشنی میں اتنا کچھ ہے کہ یہ بانٹنا چاہتا ہے کہ اب میرے لیے کوئی اور چیز اہمیت نہیں رکھتی۔ بس اس الہی محبت سے جڑے رہنے کے لیے جو آسمان سے نکلتی ہے وہ سب کچھ ہے جس کے لیے میں جی رہا ہوں، اور میں اپنے رب کی تعریف کرتا ہوں ان محبتوں کے ساتھ جو میں دعا کرتا ہوں کہ ہمیشہ کے لیے بڑھے گا۔

روح کی مافوق الفطرت فطرت مسیح کی محبت میں ایمان سے بیدار ہوتی ہے۔

ہماری دنیا نے یسوع مسیح، خدا کے مقدس بیٹے اور روشنی کو، اس کی رحمت کی گواہی کے طور پر حاصل کیا، تاکہ تمام مخلوق کو اس کے اوتار کے لیے جوابدہ ٹھہرایا جائے۔ مسیحی ہونے کے ناطے، ہم نے اپنی نجات اس کے جلال سے حاصل کی۔ اس کی روشنی تمام جسمانی اور مافوق الفطرت میں پھیلتی ہے۔ اُس کے عروج پر ایمان کے ذریعے، دونوں اب الگ نہیں رہے ہیں۔ وہ ایک ہو جاتے ہیں. اُس کا نور اُن تمام چیزوں کو چھوتا ہے جو تخلیق میں ہے، جو ہمیشہ ایک نعمت ہے، جس طرح اُس کی رحمت اور محبت ہمیشہ رہتی ہے۔
